  \begin{itemize}
    \item Can a machine learn to \highlight{create new, realistic samples}

    \pause

  \end{center}

%══════════════════════════════════════════════════════════════════

\begin{frame}{The Core Idea: Learning by Destroying}

  \begin{center}
    \begin{tikzpicture}[
      stage/.style={rounded corners=6pt, draw=TextMuted!40,
                    minimum width=2.5cm, minimum height=2.4cm,
            font=\small, text=TextPrimary, align=center,
                    inner sep=6pt},
      bigarr/.style={-{Stealth[length=3mm]}, line width=1.6pt},
      forwardarr/.style={bigarr, draw=TextMuted},
      forwardlabel/.style={font=\footnotesize, text=TextMuted},
      reversearr/.style={bigarr, draw=Highlight},
      reverselabel/.style={font=\small, text=Highlight, align=center}]

      \path[use as bounding box] (-1.2,-2.8) rectangle (11.2,1.6);

      %% Forward (destroy)
      \node[visible on=<1->, stage, fill=AccentSoft] (clean) at (0,0)
        {\textbf{Clean Image}\\[6pt]\footnotesize contains\\all the detail};
      \node[visible on=<2->, stage, fill=BgElevated] (mid) at (5,0)
        {\textbf{Noisy Image}\\[6pt]\footnotesize some detail\\remains};
      \draw[visible on=<2->, forwardarr] (clean) -- (mid)
        node[midway, above, forwardlabel]{add noise};
      \node[visible on=<3->, stage, fill=HighlightSoft] (noise) at (10,0)
        {\textbf{Pure Noise}\\[6pt]\footnotesize no information\\left};
      \draw[visible on=<3->, forwardarr] (mid) -- (noise)
        node[midway, above, forwardlabel]{add noise};
      %% Reverse (learn)
      \draw[visible on=<4->, reversearr] (noise.south) -- ++(0,-0.5) -| (clean.south);
      \node[visible on=<4->, reverselabel] at (5,-2.1)
        {Learn to reverse this process step by step};

    \end{tikzpicture}
    \pause[5]

    \vspace{6pt}
    
    \begin{block}{We Essentially Have Two Diffusion Processes}
      \begin{itemize}
        \item The \textbf{Forward Diffusion Process} that gradually adds noise.\\
        \item The \textbf{Reverse Diffusion Process} that removes noise step by step.\\
      \end{itemize}
    \end{block}
  \end{center}
\end{frame}

  \end{block}

% SLIDE 5.1 — NEW: The U-Net Architecture

\begin{frame}{The ``Brain'' of the Model: The U-Net}
  \begin{center}
    \begin{tikzpicture}[
      box/.style={rounded corners=4pt, draw=AccentMain, fill=BgElevated, thick,
                  minimum width=2.8cm, minimum height=1.5cm, align=center},
      arrow/.style={-{Stealth[length=3mm]}, thick, TextMuted, line width=1.2pt}
    ]
      % Inputs
      \node[align=center] (input) at (0,0) {
        \framebox{\includegraphics[width=2cm,height=1.6cm]{images/noised_slightly.png}}\\[2pt]
        \scriptsize Noisy Image ($x_t$)
      };
      
      % Brain
      \node[box] (unet) at (5,0) {\textbf{U-Net}\\\textbf{Architecture}\\[4pt]\scriptsize Predicts Noise ($\epsilon_\theta$)};
      
      % Outputs
      \node[align=center] (output) at (10,0) {
        \framebox{\includegraphics[width=2cm,height=1.6cm]{images/noised_fully.png}}\\[2pt]
        \scriptsize Predicted Noise ($\hat{\epsilon}$)
      };
      
      % Time step injection
      \node[align=center, text=Highlight] (time) at (5, 2) {\textbf{Time step ($t$)}};

      % Arrows
      \draw[arrow] (input) -- (unet);
      \draw[arrow] (unet) -- (output);
      \draw[arrow, Highlight, dashed] (time) -- (unet) node[midway, right] {\scriptsize injected};
    \end{tikzpicture}
  \end{center}
  
  \pause
  \vspace{10pt}
  \begin{itemize}
    \item \textbf{Input:} A noisy image and the specific time step ($t$).
    \item \textbf{Process:} The U-Net shrinks the image to understand the "global context," then expands it back, using \highlight{skip connections} to retain fine, pixel-level details.
    \item \textbf{Output:} An exact map of the noise predicted to have been added at step $t$.
  \end{itemize}
\end{frame}

\begin{frame}{Key Insight \& Why It Works}
  \begin{columns}[T]
    \column{0.48\textwidth}
      \begin{block}{Key Insight}
        \begin{itemize}
          \item We control the \highlight{noise process}
          \item Added noise is mathematically defined
          \item So the training signal is precise
          \item This makes the objective a well-defined \textit{regression problem}
        \end{itemize}
      \end{block}
      
    \pause

    \column{0.48\textwidth}
      \begin{block}{Why It Works}
        \begin{itemize}
          \item The task is simplified into \textbf{small, predictable steps}
          \item Each step only removes a tiny amount of noise
          \item Repeating simple steps reconstructs complex structure
        \end{itemize}
      \end{block}
  \end{columns}
\end{frame}

\begin{frame}{Why Diffusion Models Work So Well}
  \begin{block}{Advantages}
    \begin{itemize}
      \item \important{Stable training}
      \pause
      \item \important{High Quality Samples}
      \pause
      \item \highlight{Good Diversity} {\small (less mode collapse than GANs)}
      \pause
      \item \highlight{Flexible Conditioning} {\small (text-to-image, video-generation, audio synthesis)}
      \pause
      \item \highlight{Strong Theoretical Foundation} {\small (probabilistic modeling, score matching)}
    \end{itemize}
  \end{block}
  \pause

  \vspace{6pt}
  \begin{center}
    {In practice, diffusion models combine reliability with strong output quality.}
  \end{center}
\end{frame}

  \begin{center}
